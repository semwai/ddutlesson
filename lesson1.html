PHP

$cars = array("Volvo", "BMW", "Toyota");
/*Длину массива лучше посчитать 1 раз, а не делать 
for($i = 0; $i < count($cars); $i++)
т.к. в данном случае мы будем пересчитывать длину в каждом шаге цикла
*/
$arrlength = count($cars);

for($i = 0; $i < $arrlength; $i++) {
  echo $cars[$i];
  echo "<br>";
}
